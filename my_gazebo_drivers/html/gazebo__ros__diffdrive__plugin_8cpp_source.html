00337 // Update the controller
00338 void DiffDrivePlugin::UpdateChild()
00339 {
00340   // TODO: Step should be in a parameter of this function
00341   double wd, ws;
00342   double d1, d2;
00343   double dr, da;
00344   Time stepTime;
00345 
00346   //myIface->Lock(1);
00347 
00348   GetPositionCmd();
00349 
00350   wd = **(wheelDiamP);
00351   ws = **(wheelSepP);
00352 
00353   //stepTime = World::Instance()->GetPhysicsEngine()->GetStepTime();
00354   stepTime = Simulator::Instance()->GetSimTime() - prevUpdateTime;
00355   prevUpdateTime = Simulator::Instance()->GetSimTime();
00356 
00357   // Distance travelled by front wheels
00358   d1 = stepTime.Double() * wd / 2 * joints[RIGHT]->GetVelocity(0);
00359   d2 = stepTime.Double() * wd / 2 * joints[LEFT]->GetVelocity(0);
00360 
00361   dr = (d1 + d2) / 2;
00362   da = (d1 - d2) / ws;
00363 
00364   // Compute odometric pose
00365   odomPose[0] += dr * cos(odomPose[2]);
00366   odomPose[1] += dr * sin(odomPose[2]);
00367   odomPose[2] += da;
00368 
00369   // Compute odometric instantaneous velocity
00370   odomVel[0] = dr / stepTime.Double();
00371   odomVel[1] = 0.0;
00372   odomVel[2] = da / stepTime.Double();
00373 
00374   if (enableMotors)
00375   {
00376     joints[LEFT]->SetVelocity(0, wheelSpeed[LEFT] / (**(wheelDiamP) / 2.0));
00377 
00378     joints[RIGHT]->SetVelocity(0, wheelSpeed[RIGHT] / (**(wheelDiamP) / 2.0));
00379 
00380     joints[LEFT]->SetMaxForce(0, **(torqueP));
00381     joints[RIGHT]->SetMaxForce(0, **(torqueP));
00382   }
00383 
00384   write_position_data();
00385   publish_odometry();
00386 
00387   //myIface->Unlock();
00388 }
